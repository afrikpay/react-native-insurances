# react-native-insurances

Insurances gateway

## Installation

```sh
npm install react-native-insurances
```

## Usage

```js
import { InsuranceApp } from 'react-native-insurances';

// ...

export default function App() {
  return (
    <View>
      <InsuranceApp appToken="`your insurance system token`" lang="en" />
    </View>
  );
}
```
